        IDLE

        <<enum>>
        EXTERNAL
        INTERNAL
    }
    class StateType{
        <<enum>>
        IDLE
        MOVING_UP
        MOVING_DOWN
        DOOR_OPEN
    }
    class Request{
        -int sourceFloor
        -int destinationFloor
        -Direction direction
        -RequestType type
        -String elevatorId
        +external(src,dest)$ Request
        +internal(id,cur,dest)$ Request
    }
    class Elevator{
        -String id
        -int currentFloor
        -Direction direction
        -ElevatorState state
        -TreeSet destinations
        -int maxCapacity
        -int currentLoad
        +addDestination(floor)
        +hasDestinations() bool
        +isIdle() bool
    }

    class IdleState
    class MovingUpState
    class MovingDownState
    class DoorOpenState

    class ElevatorStateFactory{

        +selectElevator(elevators,req) Elevator

    class ElevatorRepository{
        -List elevators
        +addElevator(e)
        +findById(id) Optional
        +findAll() List
        +findActive() List
    }
    class RequestRepository{
        -Queue pendingRequests
        -List processedRequests
        +enqueue(req)
        +dequeue() Request
        +hasPending() bool
    }

    class RequestService{
        +enqueueExternalRequest(src,dest)
        +enqueueInternalRequest(id,cur,dest)
        +hasPendingRequests() bool
        +dequeueNext() Request
    }
    class ElevatorService{

        +printStatus()
    }
    class MovementService{

        +processUntilIdle()
    }
    class DispatcherService{

        +setStrategy(s)
        +dispatchNext()
        +dispatchAll()
    }

    class ExternalRequestController{
        +pressFloorButton(src,dest)
